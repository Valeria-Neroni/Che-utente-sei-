  <script>

    let step = 1;

    const score = { sprint: 0, zen: 0, critic: 0 };


    function playSound(id) {

      document.getElementById(id).play();

    }


    function nextQuestion(type) {

      playSound('clickSound');

      score[type]++;

      document.getElementById(`q${step}`).classList.add('hidden');

      step++;

      document.getElementById(`q${step}`).classList.remove('hidden');

    }


    function showResult(type) {

      playSound('successSound');

      score[type]++;

      document.getElementById(`q${step}`).classList.add('hidden');


      const max = Object.entries(score).sort((a, b) => b[1] - a[1])[0][0];

      document.getElementById('result').classList.remove('hidden');


      let title = '', text = '';

      if (max === 'sprint') {

        title = 'Lo Sprintoso ⚡';

        text = 'Parti a razzo, test after test. Ami la velocità. Lavorare con te è come essere su una zipline: diretto, senza fronzoli.';

      } else if (max === 'zen') {

        title = "L'Analitico Zen 🧘‍♂️";

        text = 'Leggi, rifletti, clicchi solo se merita. Paziente, consapevole, ma anche esigente. Le interfacce devono guadagnarsi la tua fiducia.';

      } else {

        title = 'Il Pixel Critic 👁️';

        text = 'Attenzione maniacale ai dettagli. Scrolli solo se il font ti ispira. Se noti un padding sbagliato, ti si stringe il cuore.';

      }


      document.getElementById('resultTitle').innerText = title;

      document.getElementById('resultText').innerText = text;

    }

  </script>
